30static const char *basicFragmentShaderSource = R"glsl(
31#version 450 core
32#define MAX_LIGHTS 10
33
34struct LightData {
35 vec4 position; // xyz: world-space position
36 vec4 color; // rgb: light color, w: intensity
37};
38
39layout(std140, binding = 0) uniform LightBlock {
40 int numLights;
41 LightData lights[MAX_LIGHTS];
42};
43
44uniform vec3 ambientColor;
45uniform mat4 model;
46uniform mat4 view;
47uniform mat4 projection;
48uniform vec3 viewPos;
49
50struct Material {
51 vec3 diffuseColor;
52 vec3 specularColor;
53 float specularPower;
54
55 int emissiveEnabled;
56 vec3 emissiveColor;
57};
58uniform Material material;
59
60uniform sampler2D texSampler;
61
62in vec3 FragPos;
63in vec3 Normal;
64in vec2 TexCoords;
65
66out vec4 FragColor;
67
68void main() {
69 vec3 norm = normalize(Normal);
70 vec3 viewDir = normalize(viewPos - FragPos);
71
72 vec3 ambient = ambientColor * material.diffuseColor;
73
74 vec3 diffuseTotal = vec3(0.0);
75 vec3 specularTotal = vec3(0.0);
76
77 for (int i = 0; i < numLights; i++) {
78 vec3 lightPos = lights[i].position.xyz;
79 vec3 lightColor = lights[i].color.rgb;
80 float intensity = lights[i].color.w;
81
82 vec3 lightDir = normalize(lightPos - FragPos);
83 float diff = max(dot(norm, lightDir), 0.0);
84 vec3 diffuse = diff * lightColor * material.diffuseColor;
85
86 vec3 reflectDir = reflect(-lightDir, norm);
87 float spec = 0.0;
88 if (diff > 0.0) {
89 spec = pow(max(dot(viewDir, reflectDir), 0.0), material.specularPower);
90 }
91 vec3 specular = spec * lightColor * material.specularColor;
92
93 float distance = length(lightPos - FragPos);
94 float attenuation = 1.0 / (1.0 + 0.09 * distance + 0.032 * (distance * distance));
95
96 diffuseTotal += diffuse * intensity * attenuation;
97 specularTotal += specular * intensity * attenuation;
98 }
99
100 vec3 texColor = texture(texSampler, TexCoords).rgb;
101 vec3 finalColor = (ambient + diffuseTotal + specularTotal) * texColor;
102
103 if (material.emissiveEnabled == 1) {
104 finalColor += material.emissiveColor;
105 }
106
107 FragColor = vec4(finalColor, 1.0);
108}
109)glsl";
