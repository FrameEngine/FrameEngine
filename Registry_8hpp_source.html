145 template <typename T>
146 void add_component(EntityID entity, const T &component) {
147 // Why hash? Just faster than type_index. Like O(log n) vs O(1), but there
148 // are might be hash collisions, not critical though, just restart sim.
149 size_t typeID = typeid(T).hash_code();
150
151 // If this is the first time this component type is used, create a storage
152 // bucket
153 if (componentPools.find(typeID) == componentPools.end()) {
154 componentPools[typeID] = std::make_unique<ComponentStorage<T>>();
155 }
156
157 // Store component in the bucket
158 static_cast<ComponentStorage<T> *>(componentPools[typeID].get())
159 ->add(entity, component);
160
161 // Update entity's bitmask
162 entityComponentMasks[entity].set(typeID % MAX_COMPONENTS);
163 }

171 template <typename T> bool has_component(EntityID entity) {

183 template <typename T> T &get_component(EntityID entity) {
184 size_t typeID = typeid(T).hash_code();
185 if (componentPools.find(typeID) == componentPools.end()) {
186 throw std::runtime_error("Component not found in registry!");
187 }
188 return static_cast<ComponentStorage<T> *>(componentPools[typeID].get())
189 ->get(entity);
190 }

197 template <typename T> void remove_component(EntityID entity) {
198 size_t typeID = typeid(T).hash_code();
199
200 if (componentPools.find(typeID) != componentPools.end()) {
201 static_cast<ComponentStorage<T> *>(componentPools[typeID].get())
202 ->remove(entity);
203 entityComponentMasks[entity].reset(typeID % MAX_COMPONENTS);
204 }
205 }

216 template <typename T> std::vector<EntityID> get_entities_with_component() {
217 std::vector<EntityID> result;
218 size_t typeID = typeid(T).hash_code();
219 for (auto &[entity, mask] : entityComponentMasks) {
220 if (mask.test(typeID % MAX_COMPONENTS)) {
221 result.push_back(entity);
222 }
223 }
