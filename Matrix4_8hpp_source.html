330 static Matrix4 createPerspectiveMatrix(float fov, float aspect, float near,

332 float tanHalfFOV = tan(fov * 0.5f * M_PI / 180.0f);
333
334 Matrix4 mat;
335 mat.m[0][0] = 1.0f / (aspect * tanHalfFOV);
336 mat.m[1][1] = 1.0f / tanHalfFOV;
337 mat.m[2][2] = -(far + near) / (far - near);
338 mat.m[2][3] = -1.0f;
339 mat.m[3][2] = -(2.0f * far * near) / (far - near);
340 mat.m[3][3] = 0.0f;
